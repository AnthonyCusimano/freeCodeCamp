$(document).ready(function() {
  
  //this is the user's position
  var positionLat = 0;
  var positionLong = 0;
  //this will be the user's tempurature
  var temp = 0;
  //debug
  console.log("Doc Ready");
  //taken from earlier challenge
  if (navigator.geolocation) {
    
    navigator.geolocation.getCurrentPosition(function(position) {
      positionLat = position.coords.latitude;
      positionLong = position.coords.longitude;
      
      $("#data").html("latitude: " + positionLat + "<br> longtitude: " + positionLong);
      
      $.ajax({
        url: "https://fcc-weather-api.glitch.me/api/current?lat="+positionLat+"&lon=" +positionLong,
      }).done(function(response) {
        console.log("response Get", response.main.temp);
        $( this ).addClass( "done" );
      });
    });
    
}
  
  
});
